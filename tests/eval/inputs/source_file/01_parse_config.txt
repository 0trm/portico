def parse_config(path: Path) -> Config:
    """Load a TOML config, validate against the schema, apply environment overrides."""
    raw = tomllib.loads(path.read_text())
    validated = ConfigSchema.model_validate(raw)
    overridden = apply_env_overrides(validated, prefix="ARQII_")
    return Config(
        provider=overridden.provider,
        model=overridden.model,
        cache_dir=overridden.cache_dir or DEFAULT_CACHE_DIR,
        verbose=overridden.verbose,
    )


def apply_env_overrides(cfg: ConfigSchema, *, prefix: str) -> ConfigSchema:
    out = cfg.model_copy()
    for key, value in os.environ.items():
        if not key.startswith(prefix):
            continue
        field = key[len(prefix):].lower()
        if field in cfg.model_fields:
            setattr(out, field, value)
    return out
